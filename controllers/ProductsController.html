  @Post("updateAll")
  @HttpCode(HttpStatus.OK)
  async updateAll(@Body() body, @Query() options) {
    for (let item of body.list) {
      delete item.images;

      if (item.categores) {
        let categores = [];
        for (const itm of item.categores) {
          categores.push({ id: itm });
        }
        item.categores = categores;
      }
      if (item.stones) {
        let stones = [];
        for (const itm of item.stones) {
          stones.push({ id: itm });
        }
        item.stones = stones;
      }

      this.productsService.updateById(item.id, item);
    }

    return {
      data: "done",
      returnStatus: true,
    };
  }

  @Get()
  @HttpCode(HttpStatus.OK)
  async findAll(@Query() query, @Req() req) {
    if (query.name) {
      query.name = Like("%" + query.name + "%");
    }
    let data = await this.service.find({ where: query });
    const ids = data.map((item) => item.id);

    const nowDate = new Date();
    const discountsList = await this.discountsItemsRepository.find({
      where: {
        startDate: LessThanOrEqual(nowDate),
        endDate: MoreThanOrEqual(nowDate),
        productsId: In(ids),
      },
    });

    for (let itm of data) {
      const xone = discountsList.find((el) => el.productsId.id == itm.id);
      itm.hasDiscount = false;
      itm.discount = 0;
      itm.newPrice = 0;
      if (xone) {
        const value = xone.value / 100;
        const discount = itm.price * value;
        itm.discount = xone.value;
        itm.newPrice = itm.price - discount;
        itm.hasDiscount = true;
      }
    }

    return {
      data,
      returnStatus: true,
    };
  }

  @Post("search")
  @HttpCode(HttpStatus.OK)
  async search(@Body() body, @Query() options) {
    let opt = {
      limit: Number(options.limit) || 10,
      page: Number(options.page) || 1,
    };

    let order: any = { id: "DESC" };

    let sortCol = options.sortCol;

    if (sortCol) {
      order = {};
      order[sortCol] = options.sortType;
    }

    if (body.priceFrom || body.priceTo) {
      body.price = Between(body.priceFrom, body.priceTo);
      delete body.priceFrom;
      delete body.priceTo;
    }

    let where = body;
    if (body.name) {
      const name = body.name;
      delete body.name;
      where = [
        {
          ...body,
          name: Like("%" + name + "%"),
        },
       
      ];

    }

    const data = await this.service.paginate(opt, { order, where });

    const ids = data.items.map((item) => item.id);

    const nowDate = new Date();
    const discountsList = await this.discountsItemsRepository.find({
      where: {
        startDate: LessThanOrEqual(nowDate),
        endDate: MoreThanOrEqual(nowDate),
        productsId: In(ids),
      },
    });

    for (let item of data.items) {
      const xone = discountsList.find((el) => el.productsId.id == item.id);
      item.hasDiscount = false;
      item.discount = 0;
      item.newPrice = 0;
      if (xone) {
        const value = xone.value / 100;
        const discount = item.price * value;
        item.discount = xone.value;
        item.newPrice = item.price - discount;
        item.hasDiscount = true;
      }
    }

    return {
      data,
      returnStatus: true,
    };
  }

  @Put(":id")
  @ApiCreatedResponse({
    description: "The record has been successfully updated.",
    type: Products,
  })
  @UseInterceptors(
    FilesInterceptor("images", 20, {
      storage: diskStorage({
        destination: Helper.destinationPath,
        filename: Helper.customFileName,
      }),
    })
  )
  @UseGuards(AuthGuard("jwt"))
  async update(
    @Param("id") id: number,
    @Body() body: Products,
    @UploadedFiles() images: Array<Express.Multer.File>
  ) {
    let imagesNotRemove = [];
    if (body.imagesNotRemove) {
      imagesNotRemove = JSON.parse(body.imagesNotRemove);
    }
    const list = [...imagesNotRemove];
    if (images.length > 0) {
      for (const item of images) {
        list.push(item.filename);
      }
    }

    

    delete body.imagesNotRemove;

    return await this.productsService.updateById(id, {
      ...body,
      images: JSON.stringify(list),
    });
  }

  BetweenDates = (from: Date | string, to: Date | string) =>
    Between(
      moment(typeof from === "string" ? new Date(from) : from).format(
        "YYYY-MM-DD HH:MM:SS"
      ),
      moment(typeof to === "string" ? new Date(to) : to).format(
        "YYYY-MM-DD HH:MM:SS"
      )
    );

  @Get("getIfNotDiscount/:startDate/:endDate")
  @HttpCode(HttpStatus.OK)
  async getIfNotDiscount(
    @Param("startDate") startDate,
    @Param("endDate") endDate
  ) {
    let data = await this.discountsItemsRepository.find({
      where: [
        {
          startDate: this.BetweenDates(startDate, endDate),
        },
        {
          endDate: this.BetweenDates(startDate, endDate),
        },
      ],
    });

    const ids = data.map((item) => item.productsId.id);

    let list = await this.service.find({
      where: {
        id: Not(In(ids)),
      },
    });

    return {
      data: list,
      returnStatus: true,
    };
  }
