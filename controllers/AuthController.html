import {
  Controller,
  Post,
  Body,
  HttpCode,
  HttpStatus,
  Req,
  HttpException,
  UseFilters,
  Get,
  Query,
  UseGuards,
} from "@nestjs/common";
import { AuthService } from "../services/auth.service";
import { ApiCreatedResponse, ApiTags } from "@nestjs/swagger";
import { Auth, ForgetPassword, SignUp } from "../dto/auth.dto";
import { UsersService } from "../services/users.service";

import { Users } from "../dto/users.dto";
import { ChangePassword } from "../dto/change_password.dto";
import { sendEmail } from "src/common/helpers/email";

import { getTransByKey } from "src/common/helpers/i18n";
import { HttpExceptionFilter } from "src/common/filter/http-exception.filter";
import { AuthGuard } from "@nestjs/passport";

@ApiTags("Auth")
@Controller("auth")
@UseFilters(HttpExceptionFilter)
export class AuthController {
  getTrans(key, req) {
    req.headers.language;
    return getTransByKey(key, req.headers.language);
  }
  constructor(
    private readonly authService: AuthService,
    private readonly usersService: UsersService
  ) {}

  @UseGuards(AuthGuard("facebook"))
  @Get("facebook")
  async facebook() {
    return HttpStatus.OK;
  }

  @UseGuards(AuthGuard("facebook"))
  @Get("facebook/redirect")
  async facebookRedirect(@Req() req) {
    try {
      return await this.authService.loginBytoken(
        {
          ...req.user,
          type: "facebook",
        },
        req
      );
    } catch (error) {
      throw new HttpException(error.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }

  @UseGuards(AuthGuard("google"))
  @Get("google")
  async google() {
    return HttpStatus.OK;
  }

  @UseGuards(AuthGuard("google"))
  @Get("google/redirect")
  async googleRedirect(@Req() req) {
    try {
      return await this.authService.loginBytoken(
        {
          ...req.user,
          type: "google",
        },
        req
      );
    } catch (error) {
      throw new HttpException(error.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }

  @Post("adminLogin")
  @ApiCreatedResponse({
    type: Auth,
  })
  @HttpCode(HttpStatus.OK)
  async adminLogin(@Body() body: Auth, @Req() req) {
    const data = await this.authService.adminLogin(body);
    return {
      data,
      returnStatus: true,
    };
  }

  @Post("signUp")
  @ApiCreatedResponse({
    description: "The record has been successfully created.",
    type: Users,
  })
  async create(@Body() body: SignUp, @Req() req) {
    try {
      // const password = String(this.randomNumber(8));
      await this.usersService.create({
        ...body,
        // password,
        // varifyCode: password,
      });

      const data = await this.authService.usersLogin(
        {
          email: body.email,
          password: body.password,
        },
        req
      );

      const standerdObject = {
        data,
        returnStatus: true,
      };

      sendEmail({
        to: body.email,
        templateName: "signUp",
        templateLang: req.headers.language || "en",
        templateData: {
          email: body.email,
          fullName: data.fullName,
        },
      });

      return standerdObject;
    } catch (error) {
      if (error.message.includes("Duplicate")) {
        throw new HttpException("Duplicate", HttpStatus.METHOD_NOT_ALLOWED);
      } else {
        throw new HttpException(error.message, HttpStatus.METHOD_NOT_ALLOWED);
      }
    }
  }

  @Post("login")
  @ApiCreatedResponse({
    type: Auth,
  })
  @HttpCode(HttpStatus.OK)
  async usersLogin(@Body() body: Auth, @Req() req) {
    const data = await this.authService.usersLogin(body, req);
    return {
      data,
      returnStatus: true,
    };
  }

  @Post("forgetPassword")
  @HttpCode(HttpStatus.OK)
  async forgetPassword(@Body() body: ForgetPassword, @Req() req) {
    let varifyCode = this.randomNumber(4);
    let findEmail = await this.usersService.findOne({
      email: body.email,
    });
    if (findEmail) {
      await this.usersService.update(
        {
          email: body.email,
        },
        {
          varifyCode,
        }
      );

      await sendEmail({
        to: body.email,
        templateName: "restpassword",
        templateLang: "en",
        templateData: {
          email: body.email,
          code: varifyCode,
        },
      });
      return {
        data: "varifyCodeSent",
        returnStatus: true,
      };
    } else {
      throw new HttpException(
        this.getTrans("Error, Account not found", req),
        HttpStatus.METHOD_NOT_ALLOWED
      );
    }
  }

  @Post("changePassword")
  @HttpCode(HttpStatus.OK)
  async changePassword(@Body() body: ChangePassword, @Req() req) {
    let findEmail = await this.usersService.findOne({
      where: {
        email: body.email,
        varifyCode: body.varifyCode,
      },
      select: ["varifyCode", "email", "id"],
    });

    if (findEmail && body.password == body.cpassword) {
      delete body.cpassword;
      const data = await this.usersService.updateById(findEmail.id, body);
      return {
        data,
        returnStatus: true,
      };
    } else {
      throw new HttpException(
        this.getTrans("password not equle confirmPassword", req),
        HttpStatus.METHOD_NOT_ALLOWED
      );
    }
  }

  @Post("Verification")
  @HttpCode(HttpStatus.OK)
  async Verification(@Body() body: ChangePassword, @Req() req) {
    let findEmail = await this.usersService.findOne({
      where: {
        email: body.email,
        varifyCode: body.varifyCode,
      },
      select: ["varifyCode", "email", "id"],
    });

    if (findEmail) {
      return {
        data: findEmail,
        returnStatus: true,
      };
    } else {
      throw new HttpException(
        this.getTrans("password not equle confirmPassword", req),
        HttpStatus.METHOD_NOT_ALLOWED
      );
    }
  }

  randomNumber(length) {
    let text = "";
    let possible = "012345678954871548";
    for (let i = 0; i < length; i++) {
      let sup = Math.floor(Math.random() * possible.length);
      text += i > 0 && sup == i ? "0" : possible.charAt(sup);
    }
    return Number(text);
  }
