import { UserInfo } from "../../../common/decorator/user.decorator";
import {
  Controller,
  Post,
  Body,
  Put,
  Param,
  UseGuards,
  UseFilters,
  Req,
  Get,
  UploadedFile,
  UseInterceptors,
} from "@nestjs/common";
import { Order } from "../dto/order.dto";
import { OrderService } from "../services/order.service";

import { AuthGuard } from "@nestjs/passport";
import { BaseController } from "src/common/base/baseController.controller";
import { HttpExceptionFilter } from "src/common/filter/http-exception.filter";
import { ApiCreatedResponse, ApiTags } from "@nestjs/swagger";
import { sendEmail } from "src/common/helpers/email";

@ApiTags("Order")
@Controller("order")
@UseGuards(AuthGuard("jwt"))
@UseFilters(HttpExceptionFilter)
export class OrderController extends BaseController {
  constructor(public readonly orderService: OrderService) {
    super(orderService);
  }

  @Post()
  @ApiCreatedResponse({
    description: "The record has been successfully created.",
    type: Order,
  })
  async create(@UserInfo() user, @Body() body: Order) {
    const data = await this.orderService.create(body);

    
    

    return {
      data,
      returnStatus: true,
    };
  }

  @Put(":id")
  @ApiCreatedResponse({
    description: "The record has been successfully updated.",
    type: Order,
  })
  async update(@Param("id") id: number, @Body() body: Order) {
    return await this.orderService.updateById(id, {
      ...body,
    });
  }
}
