  @Post()
  @UseGuards(AuthGuard("jwt"))
  @ApiCreatedResponse({
    description: "The record has been successfully created.",
    type: Carts,
  })
  async create(@UserInfo() user, @Body() body: any) {
    let cart = await this.cartsService.findOne({
      usersId: user.id,
      done: false,
    });
    if (cart) {
      const one = cart.cartItemsList.find(
        (el) => el.productsId.id == body.productsId
      );
      if (one) {
        const quantity = one.quantity + body.quantity;

        const nowDate = new Date();
        const xone = await this.discountsItemsRepository.findOne({
          where: {
            startDate: LessThanOrEqual(nowDate),
            endDate: MoreThanOrEqual(nowDate),
            productsId: body.productsId,
          },
        });
        let price = one.productsId.price;

        if (xone) {
          const value = xone.value / 100;
          const discount = Number(price) * value;
          price = Number(price) - discount;
        }
        const total = Number(price) * quantity;
        await this.cartItemsService.updateById(one.id, {
          quantity,
          price,
          total,
        });
      } else {
        const product = await this.productsRepository.findOne(body.productsId);

        if (!product) {
          throw new HttpException(
            "product not found",
            HttpStatus.METHOD_NOT_ALLOWED
          );
        }

        body.price = product.price;

        body.total = Number(product.price) * body.quantity;

        const nowDate = new Date();
        const xone = await this.discountsItemsRepository.findOne({
          where: {
            startDate: LessThanOrEqual(nowDate),
            endDate: MoreThanOrEqual(nowDate),
            productsId: body.productsId,
          },
        });

        if (xone) {
          const value = xone.value / 100;
          const discount = Number(product.price) * value;
          body.price = Number(product.price) - discount;
          body.total = Number(body.price) * body.quantity;
        }

        await this.cartItemsService.create({
          quantity: body.quantity,
          productsId: body.productsId,
          cartsId: cart.id,
          total: body.total,
          price: body.price,
        });
      }

      let cart2 = await this.cartsService.findOne({
        usersId: user.id,
        done: false,
      });
      let ztotal = 0;
      for (const itm of cart2.cartItemsList) {
        ztotal = ztotal + Number(itm.total);
      }

      let couponsValue = 0;

      if (cart2.couponsId) {
        if (cart2.couponsId.discountType == "value") {
          couponsValue = cart2.couponsId.value;
        }
        if (cart2.couponsId.discountType == "percenting") {
          couponsValue = Number(ztotal) * (cart2.couponsId.value / 100);
        }
      }

      const fineltotal = Number(ztotal) - Number(couponsValue);

      await this.cartsService.updateById(cart2.id, {
        couponsValue,
        total: fineltotal,
      });

      return {
        data: await { ...cart2, total: ztotal },
        returnStatus: true,
      };
    } else {
      body.total = 0;

      const product = await this.productsRepository.findOne(body.productsId);
      if (!product) {
        throw new HttpException(
          "product not found",
          HttpStatus.METHOD_NOT_ALLOWED
        );
      }
      body.price = product.price;
      body.total = product.price * body.quantity;

      const nowDate = new Date();
      const xone = await this.discountsItemsRepository.findOne({
        where: {
          startDate: LessThanOrEqual(nowDate),
          endDate: MoreThanOrEqual(nowDate),
          productsId: body.productsId,
        },
      });

      if (xone) {
        const value = xone.value / 100;
        const discount = Number(product.price) * value;
        body.price = Number(product.price) - discount;
        body.total = Number(body.price) * body.quantity;
      }

      return {
        data: await this.cartsService.create({
          total: body.total,
          cartItemsList: [body],
          usersId: user.id,
        }),
        returnStatus: true,
      };
    }
  }

          HttpStatus.METHOD_NOT_ALLOWED
        );
      }
    }
    const d = await this.cartsService.create({
      ...body,
      usersId: 8,
    });
    const one = await this.cartsService.findById(d.id);

    sendEmail({
      to: one.addressesId?.email,
      templateName: "orderCreated",
      templateLang: "ar",
      templateData: {
        name: one.addressesId?.firstName + " " + one.addressesId?.lastName,
        date: moment(new Date()).format("DD/MM/YYYY"),
        orderId: one.id,
        itemsList: one.cartItemsList,
        shippingMethodsId: one.shippingMethodsId,
        shippingPrice: one.shippingPrice,
        subTotal: one.cartItemsList
          .map((cartItem) => cartItem.total)
          .reduce((a, b) => a + b, 0),
        total: one.total,
        paymentMethod: one.paymentMethod,
        addressName:
          one.addressesId?.firstName + " " + one.addressesId?.lastName,
        addressInfo: one.addressesId?.address,
        addressApartment: one.addressesId?.apartment,
        addressCity: one.addressesId?.govesId?.name,
        addressGov: one.addressesId?.citiesId?.name,
        addressZip: one.addressesId?.zip,
        addressPhone: one.addressesId?.phone,
        addressEmail: one.addressesId?.email,
      },
    });
    return {
      data: one,
      returnStatus: true,
    };
  }
