import { UserInfo } from "../../../common/decorator/user.decorator";
import {
  Controller,
  Post,
  Body,
  UseGuards,
  UseFilters,
  Get,
  HttpCode,
  HttpStatus,
  Query,
  Param,
  HttpException,
} from "@nestjs/common";
import { SiteSetteings } from "../dto/siteSetteings.dto";
import { SiteSetteingsService } from "../services/siteSetteings.service";

import { AuthGuard } from "@nestjs/passport";

import { HttpExceptionFilter } from "src/common/filter/http-exception.filter";
import { ApiCreatedResponse, ApiTags } from "@nestjs/swagger";
import { resolve } from "path";
import * as fs from "fs";
import * as moment from "moment";

@ApiTags("Configurations")
@Controller("configurations")
// @UseGuards(AuthGuard("jwt"))
@UseFilters(HttpExceptionFilter)
export class ConfigurationsController {
  constructor() {}

  @Get("getOneEmail/:name")
  @HttpCode(HttpStatus.OK)
  async getOneEmail(@Param("name") name: string) {
    const info = JSON.parse(
      fs.readFileSync(resolve("templates", name, "info.json"), "utf8")
    );

    const subject = fs.readFileSync(
      resolve("templates", name, "subject.ejs"),
      "utf-8"
    );
    const html = fs.readFileSync(
      resolve("templates", name, "html.ejs"),
      "utf-8"
    );

    const folder = resolve("templates", "bk", name);
    const floders = await fs.readdirSync(folder);

    return {
      ...info,
      slug: name,
      backUps: floders,
      subject,
      html,
    };
  }

  @Get("getOneEmailFromBackUp/:name/:date")
  @HttpCode(HttpStatus.OK)
  async getOneEmailFromBackUp(
    @Param("name") name: string,
    @Param("date") date: string
  ) {
    const info = JSON.parse(
      fs.readFileSync(resolve("templates", name, "info.json"), "utf8")
    );

    const subject = fs.readFileSync(
      resolve("templates", "bk", name, date, "subject.ejs"),
      "utf-8"
    );
    const html = fs.readFileSync(
      resolve("templates", "bk", name, date, "html.ejs"),
      "utf-8"
    );

    return {
      ...info,
      slug: name,
      subject,
      html,
    };
  }

  @Post("updateEmail/:name")
  @ApiCreatedResponse({
    description: "The record has been successfully created.",
    type: SiteSetteings,
  })
  async updateEmail(@Param("name") name: string, @Body() body: any) {
    try {
      const date = moment(new Date()).format("DD-MM-YYYY_HH:mm:ss");

      // await fs.mkdirSync(resolve("templates", "bk", name));
      await fs.mkdirSync(resolve("templates", "bk", name, date), {
        recursive: true,
      });
      await fs.copyFileSync(
        resolve("templates", name, "subject.ejs"),
        resolve("templates", "bk", name, date, "subject.ejs")
      );
      await fs.copyFileSync(
        resolve("templates", name, "html.ejs"),
        resolve("templates", "bk", name, date, "html.ejs")
      );

      await fs.writeFileSync(
        resolve("templates", name, "subject.ejs"),
        body.subject
      );
      await fs.writeFileSync(resolve("templates", name, "html.ejs"), body.html);

      return {
        done: true,
      };
    } catch (e) {
      throw new HttpException(e.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }

  @Get("geti18nEmail")
  @ApiCreatedResponse({
    description: "The record has been successfully created.",
    type: SiteSetteings,
  })
  async geti18nEmail() {
    try {
      const en = JSON.parse(
        fs.readFileSync(resolve("templates", "i18n", "en.json"), "utf8")
      );
      const ar = JSON.parse(
        fs.readFileSync(resolve("templates", "i18n", "ar.json"), "utf8")
      );
      const list = [];
      for (const key in en) {
        let arvalue = "";
        if (Object.prototype.hasOwnProperty.call(ar, key)) {
          arvalue = ar[key];
        }
        list.push({
          key,
          en: en[key],
          ar: arvalue,
        });
      }
      return list;
    } catch (e) {
      throw new HttpException(e.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }

  @Post("updatei18nEmail")
  @ApiCreatedResponse({
    description: "The record has been successfully created.",
    type: SiteSetteings,
  })
  async updatei18nEmail(@Body() body: any) {
    try {
      let en = {};
      let ar = {};
      for (const item of body) {
        en[item.key] = item.en;
        ar[item.key] = item.ar;
      }

      fs.writeFileSync(resolve("templates", "i18n", "en.json"), JSON.stringify(en), "utf8");

      fs.writeFileSync(resolve("templates", "i18n", "ar.json"), JSON.stringify(ar), "utf8");
      return {
        done: true,
      };
    } catch (e) {
      throw new HttpException(e.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }
}
