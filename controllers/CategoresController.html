import { CategoresService } from "../services/categores.service";

@ApiTags("Categores")
@Controller("categores")

@UseFilters(HttpExceptionFilter)

  constructor(
    public readonly categoresService: CategoresService,

  @Post("getData")
  @HttpCode(HttpStatus.OK)
  async getData(@Body() body, @Query() options) {
    options.limit = Number(options.limit) || 10;
    options.page = Number(options.page) || 1;

    const { path, sub } = await this.categoresService.getBySlug(options.slug);

    let order: any = { id: "DESC" };

    let sortCol = options.sortCol;

    if (sortCol) {
      order = {};
      order[sortCol] = options.sortType;
    }

    if (body.priceFrom || body.priceTo) {
      body.price = Between(body.priceFrom, body.priceTo);
      delete body.priceFrom;
      delete body.priceTo;
    }

    let where = body;
    if (body.name) {
      where = [
        {
          ...body,
          name: Like("%" + body.name + "%"),
        },
        {
          ...body,
          nameEn: Like("%" + body.name + "%"),
        },
      ];
    }

    // IN (:...roles)

    const items = await this.productsService.paginate(
      {
        limit: options.limit,
        page: options.page,
      },
      {
        order,
        relations: ['categores'],
        where:(qb: SelectQueryBuilder<ProductsEntity>) => {
          qb.where('categoresId IN (:...categoryId)', {categoryId: sub})
        }
        
        //  {
          
        //   categores: In(sub),
        //   ...where,
        // },
      }
    );

    const ids = items.items.map((item) => item.id);

    const nowDate = new Date();
    const discountsList = await this.discountsItemsRepository.find({
      where: {
        startDate: LessThanOrEqual(nowDate),
        endDate: MoreThanOrEqual(nowDate),
        productsId: In(ids),
      },
    });

    for (let item of items.items) {
      const xone = discountsList.find((el) => el.productsId.id == item.id);
      item.hasDiscount = false;
      item.discount = 0;
      item.newPrice = 0;
      if (xone) {
        const value = xone.value / 100;
        const discount = item.price * value;
        item.discount = xone.value;
        item.newPrice = item.price - discount;
        item.hasDiscount = true;
      }
    }
    return {
      data: {
        path,

        list: items,
      },
      returnStatus: true,
    };
  }

  @Post("geAllCounter")
  @HttpCode(HttpStatus.OK)
  async geAllCounter(@Body() body, @Query() options) {
    const cat: any = await this.categoresService.find({
      where: {
        parent: null,
      },
    });

    const list = [];
    for (const item of cat) {
      const { path, sub } = await this.categoresService.getBySlug(item.slug);

      const itemsCount = await this.productsService.count({
        categores: In(sub),
      });
      list.push({
        id: item.id,
        name: item.name,
        nameEn: item.nameEn,
        image: item.image,
        itemsCount,
      });
    }

    return {
      data: list,
      returnStatus: true,
    };
  }
