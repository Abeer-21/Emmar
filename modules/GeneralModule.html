import { Module } from "@nestjs/common";
import { TypeOrmModule } from "@nestjs/typeorm";

import { TrademarksEntity } from "./entity/trademarks.entity";
import { TrademarksController } from "./controllers/trademarks.controller";
import { TrademarksService } from "./services/trademarks.service";
import { SiteSetteingsEntity } from "./entity/siteSetteings.entity";
import { SiteSetteingsController } from "./controllers/siteSetteings.controller";
import { SiteSetteingsService } from "./services/siteSetteings.service";
import { MailchimpEntity } from "./entity/mailchimp.entity";
import { MailchimpController } from "./controllers/mailchimp.controller";
import { MailchimpService } from "./services/mailchimp.service";
import { ConfigurationsController } from "./controllers/configurations.controller";
import { HomeSetteingsEntity } from "./entity/homeSetteings.entity";
import { HomeSetteingsController } from "./controllers/homeSetteings.controller";
import { HomeSetteingsService } from "./services/homeSetteings.service";
import { GalleriesEntity } from "./entity/galleries.entity";
import { GalleriesController } from "./controllers/galleries.controller";
import { GalleriesService } from "./services/galleries.service";

@Module({
  imports: [
    TypeOrmModule.forFeature([
      TrademarksEntity,
      SiteSetteingsEntity,
      MailchimpEntity,
      HomeSetteingsEntity,
      GalleriesEntity,
    ]),
  ],
  controllers: [
    TrademarksController,
    SiteSetteingsController,
    MailchimpController,
    ConfigurationsController,
    HomeSetteingsController,
    GalleriesController,
  ],
  providers: [
    TrademarksService,
    SiteSetteingsService,
    MailchimpService,
    HomeSetteingsService,
    GalleriesService,
  ],
})
export class GeneralModule {}
