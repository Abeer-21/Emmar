import { HttpException, HttpStatus, Injectable } from "@nestjs/common";
import { BaseService } from "src/common/base/baseService.service";
import { IPaginationOptions, paginate } from "nestjs-typeorm-paginate";

import { InjectRepository } from "@nestjs/typeorm";
import { Repository } from "typeorm";
import { ProductsEntity } from "src/modules/products/entity/products.entity";
import { CartItemsEntity } from "src/modules/carts/entity/cart_items.entity";
import { PurchasingItemsEntity } from "src/modules/products/entity/purchasingItems.entity";
import { PurchasingEntity } from "src/modules/products/entity/purchasing.entity";
import { CartsEntity } from "src/modules/carts/entity/carts.entity";
@Injectable()
export class ReportsService {
  constructor(
    @InjectRepository(ProductsEntity)
    private readonly productsRepository: Repository<ProductsEntity>,

    @InjectRepository(CartItemsEntity)
    private readonly cartItemsRepository: Repository<CartItemsEntity>,

    @InjectRepository(PurchasingItemsEntity)
    private readonly purchasingItemsRepository: Repository<
      PurchasingItemsEntity
    >,
    @InjectRepository(PurchasingEntity)
    private readonly purchasingRepository: Repository<PurchasingEntity>,

    @InjectRepository(CartsEntity)
    private readonly cartsRepository: Repository<CartsEntity>
  ) {}

  async getItems(options: IPaginationOptions, body) {
    try {
      const list = await paginate(this.productsRepository, options, body);
      for (let item of list.items) {
        const soldQuantity = await this.cartItemsRepository.find({
          where: { productsId: item.id },
          select: ["id", "productsId", "quantity"],
        });
        item.soldQuantity = soldQuantity
          .map((itm) => itm.quantity)
          .reduce((acc, item) => acc + item, 0);

        const purchasedQuantity = await this.purchasingItemsRepository.find({
          where: { productsId: item.id },
          select: ["id", "productsId", "quantity"],
        });

        item.purchasedQuantity = purchasedQuantity
          .map((itm) => itm.quantity)
          .reduce((acc, item) => acc + item, 0);
      }
      return list;
    } catch (e) {
      throw new HttpException(e.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }

  async getPurchasingItems(options: IPaginationOptions, body) {
    try {
      const item = await paginate(
        this.purchasingItemsRepository,
        options,
        body
      );
      for (let it of item.items) {
        it.purchasingId = await this.purchasingRepository.findOne({
          where: { id: it.purchasingId },
          loadEagerRelations: false,
        });
      }
      return item;
    } catch (e) {
      throw new HttpException(e.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }
  async getSold(options: IPaginationOptions, body) {
    try {
      const item = await paginate(this.cartItemsRepository, options, body);
      for (let it of item.items) {
        it.cartsId = await this.cartsRepository.findOne({
          where: { id: it.cartsId },
          loadEagerRelations: false,
        });
      }
      return item;
    } catch (e) {
      throw new HttpException(e.message, HttpStatus.METHOD_NOT_ALLOWED);
    }
  }

    }
  }
}
