import { Entity, Column, PrimaryGeneratedColumn, OneToMany } from "typeorm";
import * as bcrypt from "bcryptjs";
import { AddressesEntity } from "./addresses.entity";
import { WishlistEntity } from "src/modules/carts/entity/wishlist.entity";
import { CartsEntity } from "src/modules/carts/entity/carts.entity";

@Entity({ name: "users" })
export class UsersEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ nullable: true })
  fullName: string;

  @Column({ unique: true })
  email: string;

  @Column({ select: false })
  password: string;

  @Column({ nullable: true })
  varifyCode: string;

  @Column({ nullable: true })
  facebookId: string;

  @Column({ nullable: true })
  googleId: string;

  @Column({ nullable: true })
  fname: string;

  @Column({ nullable: true })
  lname: string;

  @Column({ nullable: true })
  country: string;

  @Column({ nullable: true })
  city: string;

  @Column({ nullable: true })
  address: string;

  @Column({ nullable: true })
  unit: string;

  @Column({ nullable: true })
  zip: string;

  @Column({ nullable: true })
  phone: string;

  @Column({
    default: 0,
    nullable: true,
    type: "decimal",
    precision: 20,
    scale: 2,
  })
  wallet: number;

  @Column({
    nullable: true,
    default: "Customer",
  })
  userType: string;

  @Column({ type: "boolean" })
  newsletter: boolean;

  @OneToMany(
    () => AddressesEntity,
    (item) => item.usersId,
    {
      onUpdate: "CASCADE",
      onDelete: "CASCADE",
      eager: true,
    }
  )
  addressesList: AddressesEntity[];

  @OneToMany(
    () => WishlistEntity,
    (item) => item.usersId,
    {
      onUpdate: "CASCADE",
      onDelete: "CASCADE",
      eager: true,
    }
  )
  wishlistList: WishlistEntity[];

  @OneToMany(
    () => CartsEntity,
    (item) => item.usersId,
    {
      onUpdate: "CASCADE",
      onDelete: "CASCADE",
    }
  )
  cartsList: CartsEntity[];

  @Column({ type: "timestamp", default: () => "CURRENT_TIMESTAMP" })
  createdAt: string;

  @Column({ type: "timestamp", default: () => "CURRENT_TIMESTAMP" })
  updatedAt: string;

  async comparePassword(attempt: string): Promise<boolean> {
    return await bcrypt.compare(attempt, this.password);
  }
}
