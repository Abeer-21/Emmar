import { AddressesEntity } from "src/modules/users/entity/addresses.entity";
import { UsersEntity } from "src/modules/users/entity/users.entity";
import {
  Entity,
  Column,
  PrimaryGeneratedColumn,
  OneToMany,
  ManyToOne,
  Unique,
} from "typeorm";
import { CartItemsEntity } from "./cart_items.entity";
import { ShippingMethodsEntity } from "./shippingMethods.entity";
import { CouponsEntity } from "./coupons.entity";

@Entity({ name: "carts" })
export class CartsEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @ManyToOne(
    () => UsersEntity,
    (item) => item.cartsList,
    {
      eager: true,
    }
  )
  usersId: UsersEntity;

  @Column({ default: "transfer" })
  paymentMethod: string;

  @Column({ nullable: true })
  order_id: string;

  @Column({ nullable: true })
  checkout_id: string;

  @Column({ nullable: true })
  smsaexpressNo: string;

  @Column({ nullable: true })
  pdfInvoice: string;

  @Column({
    // nullable: true,
    default: "pending",
    // enum: ["pending", "shipping", "inDelivery", "delivered"],
  })
  orderStatus: string;

  @Column({
    default: 0,
    nullable: true,
    type: "decimal",
    precision: 20,
    scale: 2,
  })
  shippingPrice: number;

  @Column({
    default: 0,
    nullable: true,
    type: "decimal",
    precision: 20,
    scale: 2,
  })
  couponsValue: number;

  @Column({
    default: 0,
    nullable: true,
    type: "decimal",
    precision: 20,
    scale: 2,
  })
  total: number;

  @Column({ type: "boolean", default: false })
  giftPackaging: boolean;

  @Column({ type: "boolean", default: false })
  done: boolean;

  @ManyToOne(
    () => AddressesEntity,
    (item) => item.cartsList,
    {
      eager: true,
      cascade: true,
    }
  )
  addressesId: AddressesEntity;

  @ManyToOne(
    () => CouponsEntity,
    (item) => item.cartsList,
    {
      eager: true,
    }
  )
  couponsId: CouponsEntity;

  @ManyToOne(
    () => ShippingMethodsEntity,
    (item) => item.cartsList,
    {
      eager: true,

      onUpdate: "CASCADE",
      onDelete: "CASCADE",
    }
  )
  shippingMethodsId: ShippingMethodsEntity;

  @OneToMany(
    () => CartItemsEntity,
    (item) => item.cartsId,
    {
      eager: true,

      onUpdate: "CASCADE",
      onDelete: "CASCADE",
      cascade: true,
    }
  )
  cartItemsList: CartItemsEntity[];

  @Column({ type: "timestamp", default: () => "CURRENT_TIMESTAMP" })
  createdAt: string;

  @Column({ type: "timestamp", default: () => "CURRENT_TIMESTAMP" })
  updatedAt: string;
}
