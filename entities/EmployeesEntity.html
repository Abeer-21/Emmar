import { StoresEntity } from "src/modules/users/entity/stores.entity";
import {
  Entity,
  Column,
  PrimaryGeneratedColumn,
  ManyToOne,
  OneToMany,
} from "typeorm";
import { PayrollItemsEntity } from "./payroll_items.entity";
import { FollowupEntity } from "./followup.entity";
import { LeaveEntity } from "./leave.entity";
import { AlarmsEntity } from "./alarms.entity";
import { OrderEntity } from "./order.entity";
import { InvestigationEntity } from "./investigation.entity";
import { BranchesEntity } from "./branches.entity";

@Entity({ name: "employees" })
export class EmployeesEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column()
  code: string;

  @Column()
  fullName: string;

  @Column({ nullable: true })
  shortName: string;

  @Column({ nullable: true })
  phone: string;

  @Column({ nullable: true })
  email: string;

  @Column({ nullable: true })
  mobileNumber: string;

  @Column({ nullable: true })
  IDNumber: string;

  @Column({ nullable: true })
  placeOfBirth: string;

  @Column({ nullable: true })
  job: string;

  @Column({ nullable: true })
  nationality: string;

  @Column({ nullable: true })
  joiningDate: string;

  @Column({ nullable: true })
  expiryDate: string;

  @Column({ nullable: true })
  passportNumber: string;

  @Column({ nullable: true })
  passportExpiryDate: string;

  @Column({ nullable: true })
  dateOfBirth: string;

  @Column({ nullable: true })
  image: string;


  @Column({ nullable: true })
  pdf: string;

  @Column({ nullable: true })
  insuranceNumber: string;

  @Column({ nullable: true })
  maritalStatus: string;

  @Column({ nullable: true })
  familyNumber: string;

  @Column({ nullable: true })
  durationTrial: string;

  @Column({ nullable: true })
  endReason: string;

  @Column({ nullable: true })
  endingDate: string;

  @Column({ nullable: true })
  nameEn: string;

  @Column({ nullable: true })
  nationalityEn: string;

  @Column({ nullable: true })
  jobEn: string;


  @Column({ nullable: true })
  contractType: string;
  

  @Column({ nullable: true })
  durationTrialEnd: string;

  @Column({ type: "int", default: 0, nullable: true })
  vacationsBalance: string;

  @Column({ type: "int", default: 0, nullable: true })
  vacationsBalanceUsed: string;

  @Column({ type: "int", default: 0, nullable: true })
  vacationsBalanceNotUsed: string;

  @Column({ nullable: true })
  professionalLicenseNumber: string;

  @Column({ nullable: true })
  professionalLicenseNumberEnd: string;

  @Column({ type: "int", default: 0, nullable: true })
  basicSalary: number;

  @Column({ type: "int", default: 0, nullable: true })
  housingAllowance: number;

  @Column({ type: "int", default: 0, nullable: true })
  otherAllowances: number;

  @Column({ type: "int", default: 0, nullable: true })
  totalSalary: number;

  @Column({ nullable: true })
  numberInBusinessOffice: number;

  @Column({ nullable: true })
  bankAccount: string;

  @Column({ nullable: true })
  bankAccountNumber: string;

  @Column({ type: "text", nullable: true })
  notes: string;

  @Column({ type: "boolean", default: false })
  active: boolean;

  @Column({ type: "boolean", default: false })
  guarantee: boolean;

  @ManyToOne(
    () => BranchesEntity,
    (item) => item.branchesList,
    {
      eager: true,
    }
  )
  branchesId: BranchesEntity;

  @OneToMany(
    () => PayrollItemsEntity,
    (item) => item.employeesId,
    {}
  )
  payrollList: PayrollItemsEntity[];

  @OneToMany(
    () => FollowupEntity,
    (item) => item.employeesId,
    {}
  )
  followupList: FollowupEntity[];

  @OneToMany(
    () => LeaveEntity,
    (item) => item.employeesId,
    {}
  )
  leaveList: LeaveEntity[];

  @OneToMany(
    () => AlarmsEntity,
    (item) => item.employeesId,
    {}
  )
  alarmsList: AlarmsEntity[];

  @OneToMany(
    () => OrderEntity,
    (item) => item.employeesId,
    {}
  )
  orderList: OrderEntity[];

  @OneToMany(
    () => InvestigationEntity,
    (item) => item.employeesId,
    {}
  )
  investigationList: InvestigationEntity[];

  @Column({ type: "timestamp", default: () => "CURRENT_TIMESTAMP" })
  createdAt: string;

  @Column({ type: "timestamp", default: () => "CURRENT_TIMESTAMP" })
  updatedAt: string;
}
